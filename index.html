totalpedido 1

Calcula el total de un pedido sumando (unitprice * quantity).

-- Calcula total de pedido
CREATE FUNCTION dbo.totalpedido(@orderid INT)
RETURNS MONEY
AS
BEGIN
  DECLARE @total MONEY
  SELECT @total = SUM(unitprice * quantity)
  FROM [order details]
  WHERE orderid = @orderid
  RETURN ISNULL(@total, 0)
END
-- Prueba: SELECT dbo.totalpedido(10248)

nombreempleado 2

Devuelve nombre completo de empleado concatenando firstname y lastname.

-- Devuelve nombre completo de empleado
CREATE FUNCTION dbo.nombreempleado(@employeeid INT)
RETURNS VARCHAR(100)
AS
BEGIN
  DECLARE @nombre VARCHAR(100)
  SELECT @nombre = firstname + ' ' + lastname
  FROM employees
  WHERE employeeid = @employeeid
  RETURN ISNULL(@nombre, 'no encontrado')
END
-- Prueba: SELECT dbo.nombreempleado(1)

CREATE FUNCTION dbo.pedidoscliente(@customerid NCHAR(5))
RETURNS TABLE
AS
RETURN
(
  SELECT orderid, orderdate, shipcountry
  FROM orders
  WHERE customerid = @customerid
)

Calcula cantidad total vendida (qty) para un título específico.

-- Calcula total vendido por título
CREATE FUNCTION dbo.fn_TotalVentasPorTítulo(@title_id VARCHAR(6))
RETURNS INT
AS
BEGIN
  DECLARE @total INT
  SELECT @total = SUM(qty)
  FROM sales
  WHERE title_id = @title_id
  RETURN ISNULL(@total, 0)
END
-- Prueba: SELECT dbo.fn_TotalVentasPorTítulo('PS3333')

-- Cuenta títulos por editorial
CREATE FUNCTION dbo.fn_CantidadTítulosPorEditorial(@pub_id VARCHAR(4))
RETURNS INT
AS
BEGIN
  DECLARE @cantidad INT
  SELECT @cantidad = COUNT(*)
  FROM titles
  WHERE pub_id = @pub_id
  RETURN ISNULL(@cantidad, 0)
END
-- Prueba: SELECT dbo.fn_CantidadTítulosPorEditorial('0736')

fn_AutoresConTitulos 6

Devuelve tabla con autores y sus libros (JOIN entre 3 tablas).

-- Muestra autores con sus libros
CREATE FUNCTION dbo.fn_AutoresConTitulos()
RETURNS TABLE
AS
RETURN
(
  SELECT
    a.au_lname + ' ' + a.au_fname AS Autor,
    t.title AS Titulo
  FROM authors a
  JOIN titleauthor ta ON a.au_id = ta.au_id
  JOIN titles t ON ta.title_id = t.title_id
)
-- Prueba: SELECT * FROM dbo.fn_AutoresConTitulos()

AS

END
